import java.util.Scanner;

public class IdadeAlturaPeso {
    public static void main(String[] args) {
        Scanner ler = new Scanner(System.in);

        double somaAltura = 0, mediaAltura = 0;
        int superior = 0, quantidadeAltura = 0;
        int pessoasPeso = 0;
        int velho = 0, novo = 0;

        for (int i = 1; i <= 25; i++) {
            System.out.println("Pessoa " + i + ":");
            System.out.print("Digite a idade: ");
            int idade = ler.nextInt();

            System.out.print("Digite a altura (em metros): ");
            double altura = ler.nextDouble();

            System.out.print("Digite o peso (em kg): ");
            double peso = ler.nextDouble();

           
            if (i == 1) {
                velho = idade;
                novo = idade;
            }

            if (idade > 50) {
                superior++;
            }

            if (idade >= 10 && idade <= 20) {
                somaAltura += altura;
                quantidadeAltura++;
            }

            if (peso < 40) {
                pessoasPeso++;
            }

            
            if (idade > velho) {
                velho = idade;
            }
            if (idade < novo) {
                novo = idade;
            }

            System.out.println();
        }

        if (quantidadeAltura > 0) {
            mediaAltura = somaAltura / quantidadeAltura;
        }

    
        double porcentagem = (pessoasPeso * 100.0) / 25.0;

        System.out.println("Quantidade de pessoas com idade superior a 50 anos: " + superior);

        System.out.printf("MÃ©dia das alturas das pessoas com idade entre 10 e 20 anos: %.2f metros\n", mediaAltura);

        System.out.printf("Porcentagem de pessoas com peso inferior a 40 kg: %.2f%%\n", porcentagem);

        System.out.println("Idade da pessoa mais velha: " + velho);
        System.out.println("Idade da pessoa mais nova: " + novo);

        ler.close();
    }
}
