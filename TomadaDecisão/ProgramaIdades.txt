import java.util.Scanner;

public class IdadePessoas {

    public static void main(String[] args) {

        Scanner ler = new Scanner(System.in);

        String nome1, nome2;

        int idade1, idade2;

        System.out.print("Digite o nome da primeira pessoa: ");
        nome1 = ler.nextLine();

        System.out.print("Digite a idade da primeira pessoa: ");
        idade1 = ler.nextInt();


        System.out.print("Digite o nome da segunda pessoa: ");
        nome2 = ler.nextLine();

        System.out.print("Digite a idade da segunda pessoa: ");
        idade2 = ler.nextInt();



        if (idade1 > idade2) {

            System.out.println("A mais velha é " + nome1 + " e tem " + idade1 + " anos.");

            System.out.println("A mais nova é " + nome2 + " e tem " + idade2 + " anos.");

        } else if (idade2 > idade1) {

            System.out.println("A mais velha é " + nome2 + " e tem " + idade2 + " anos.");

            System.out.println("A mais nova é " + nome1 + " e tem " + idade1 + " anos.");

        } else {

            System.out.println(nome1 + " e " + nome2 + " têm a mesma idade.");

        }

        ler.close();

    }

}
