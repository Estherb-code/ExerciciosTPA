import java.util.Scanner;

public class SalarioComparacao {

    public static void main(String[] args) {

        Scanner ler = new Scanner(System.in);

        double salarioMinimo, salarioFuncionario, diferenca;

       

        System.out.print("Digite o valor do salário mínimo: ");
        salarioMinimo = ler.nextDouble();

        System.out.print("Digite o salário do funcionário: ");
        salarioFuncionario = ler.nextDouble();

  
        if (salarioFuncionario > salarioMinimo) {

            diferenca = salarioFuncionario - salarioMinimo;

            System.out.println("O funcionário recebe acima do salário mínimo!");

            System.out.println("Diferença: R$ " + diferenca);

        } 

        else if (salarioFuncionario < salarioMinimo) {

            diferenca = salarioMinimo - salarioFuncionario;

            System.out.println("O funcionário recebe abaixo do salário mínimo!");

            System.out.println("Diferença: R$ " + diferenca);

        } 

        else {

            System.out.println("O funcionário recebe o mesmo valor que o salário mínimo!");

        }

        ler.close();

    }

}
