import java.util.Scanner;

class Tabuada {

    public static void main(String[] args) {

        Scanner ler = new Scanner(System.in);

        int i, nu, tabu;

        i = 1;

        System.out.println("Digite o n√∫mero que deseja a tabuada:");

        nu = ler.nextInt();

        while (i <= 10) {

            tabu = nu * i;

            System.out.println("Tabuada desejada: " + tabu);

            i++;

        }

        ler.close();

    }

}
