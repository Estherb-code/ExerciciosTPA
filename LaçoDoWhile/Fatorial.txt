import java.util.Scanner;
public class Fatorial {
   public static void main(String[] args) {
       Scanner ler = new Scanner(System.in);
       
       int i = 1;
       int n;
       int multi = 1;
       
       System.out.println("Apresente o numero:");
       n = ler.nextInt();
       do {
           multi = multi * i;
           i++;
       } while (i <= n);
       ler.close();
       
       System.out.println("A fatoração é:" + multi);
   }
}
